// Submits audio to AWS Transcribe, polls for completion, returns transcript.
// In mock mode returns realistic sample transcripts immediately.
 
const {
  TranscribeClient,
  StartTranscriptionJobCommand,
  GetTranscriptionJobCommand,
} = require('@aws-sdk/client-transcribe');
const { v4: uuidv4 } = require('uuid');
const axios   = require('axios');
const config  = require('../config');
const logger  = require('../utils/logger');
const { withRetry } = require('../utils/retry');
 
// ─── Real client ─────────────────────────────────────────────
let transcribeClient = null;
function getClient() {
  if (!transcribeClient) {
    transcribeClient = new TranscribeClient({
      region: config.aws.region,
      credentials: config.aws.accessKeyId
        ? { accessKeyId: config.aws.accessKeyId, secretAccessKey: config.aws.secretAccessKey }
        : undefined,
    });
  }
  return transcribeClient;
}

// ─── Mock implementation ─────────────────────────────────────
async function mockTranscribe(audioKey) {
  logger.debug('[MOCK Transcribe] Simulating transcription', { audioKey });
  await new Promise(r => setTimeout(r, 800)); // simulate processing time
 
  // Pick a transcript deterministically based on the audio key
  const idx = Math.abs(audioKey.split('').reduce((a, c) => a + c.charCodeAt(0), 0)) % MOCK_TRANSCRIPTS.length;
  const sample = MOCK_TRANSCRIPTS[idx];
 
  return {
    transcript: sample.text,
    confidence: sample.confidence,
    language:   'en-US',
    job_name:   `mock-job-${uuidv4().slice(0, 8)}`,
    mock:       true,
  };
}
 
// ─── Real implementation ─────────────────────────────────────
 
/** Poll for a Transcribe job until COMPLETED or FAILED */
async function pollJob(client, jobName, maxWaitMs = 300_000, intervalMs = 5_000) {
  const deadline = Date.now() + maxWaitMs;
 
  while (Date.now() < deadline) {
    const { TranscriptionJob: job } = await client.send(
      new GetTranscriptionJobCommand({ TranscriptionJobName: jobName })
    );
 
    const status = job.TranscriptionJobStatus;
    logger.debug('Transcribe poll', { jobName, status });
 
    if (status === 'COMPLETED') {
      // Fetch the transcript JSON from the result URL
      const url  = job.Transcript.TranscriptFileUri;
      const resp = await axios.get(url);
      const results = resp.data.results;
      const transcript = results.transcripts.map(t => t.transcript).join(' ');
      const confidence = parseFloat(
        results.items
          .filter(i => i.alternatives?.[0]?.confidence)
          .reduce((s, i) => s + parseFloat(i.alternatives[0].confidence), 0) /
        results.items.filter(i => i.alternatives?.[0]?.confidence).length || 0
      );
      return { transcript, confidence: Math.round(confidence * 100), language: job.LanguageCode, job_name: jobName };
    }
 
    if (status === 'FAILED') {
      throw new Error(`Transcribe job failed: ${job.FailureReason}`);
    }
 
    await new Promise(r => setTimeout(r, intervalMs));
  }
 
  throw new Error(`Transcribe job timed out after ${maxWaitMs / 1000}s`);
}
 
/**
 * Transcribe an audio file stored in S3.
 * @param {string} audioKey  - S3 object key
 * @param {string} callId    - Used as part of the job name
 */
async function transcribeAudio(audioKey, callId) {
  if (config.useMockAws) return mockTranscribe(audioKey);
 
  const client  = getClient();
  const jobName = `govai-${callId}-${Date.now()}`;
  const s3Uri   = `s3://${config.aws.s3Bucket}/${audioKey}`;
 
  logger.info('Starting Transcribe job', { jobName, s3Uri });
 
  await withRetry(
    () => client.send(new StartTranscriptionJobCommand({
      TranscriptionJobName: jobName,
      Media:                { MediaFileUri: s3Uri },
      MediaFormat:          audioKey.endsWith('.mp3') ? 'mp3' : 'wav',
      LanguageCode:         'en-US',
      Settings: {
        ShowSpeakerLabels:    true,
        MaxSpeakerLabels:     2,
        ShowAlternatives:     false,
      },
    })),
    config.retry.maxRetries,
    config.retry.delayMs,
    'StartTranscriptionJob'
  );
 
  return pollJob(client, jobName);
}
 
module.exports = { transcribeAudio };
